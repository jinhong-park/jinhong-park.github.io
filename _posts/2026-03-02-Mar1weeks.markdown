    if (localStorage.getItem('detailsOpen') === 'true') {
      details.setAttribute('open', '');
    }
    
    // Save the state when the details element is toggled
    details.addEventListener('toggle', () => {
      localStorage.setItem('detailsOpen', details.open);
    });
  });
</script>
